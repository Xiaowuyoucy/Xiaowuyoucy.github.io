# 枚举的增强

```c++
#include<iostream>

using namespace std;

enum Son {
	xiaom,
	xiaoh,
	xiaol,
	xiaolan,
};

int main(int argc, char* argv[])
{
    //可以省略enum关键字不写
	Son house = xiaom;
	//c++中,枚举变量只能用枚举常量来赋值
	//house = 1;

	cout << house << endl;

	cin.get();
	return 0;
}



```
