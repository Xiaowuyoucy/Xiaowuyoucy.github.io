# 类模板的基本语法

类模板必须显式指定类型

```c
#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
#include<string>

using namespace std;

template<class T>
class Person {
	
private:
	T mA;


public:
	Person(){}
	Person(T a) {
		mA = a;
	}
	void show() {
		cout << mA << endl;
	}

};

void test01() {
	//类模板必须显示调用
	Person<string> str("123");
	str.show();
}


int main(char *argv[], int argc)
{
	test01();

	return 0;
}

```
