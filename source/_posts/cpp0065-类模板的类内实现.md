# 类模板的类内实现

```cpp
#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
#include<string>

using namespace std;

template<class T1,class T2>
class Person {
private:
	T1 mName;
	T2 mAge;

public:
	Person(){}
	Person(T1 name,T2 age):mName(name),mAge(age) {}
	void show() {
		cout << mName << ":" << mAge << "岁" << endl;
	}

};

void test01() {
	Person<string, int> p("小明",18);
	p.show();
}

int main(char *argv[], int argc)
{
	test01();

	return 0;
}

```
