# 对象返回自身

```c++

#include<iostream>


using namespace std;

class Test
{
public:
	Test();
	~Test();
	const Test & print() const {
		cout << a << endl;
		cout << b << endl;
		cout << c << endl;
		cout << "====================" << endl;
		return *this; //返回自己本身
	}
private:
	int a;
	int b;
	int c;
};

Test::Test()
{
	a = 10;
	b = 20;
	c = 30;
}

Test::~Test()
{
}

int main(char *argv[], int argc)
{
	Test t1;
	t1.print().print();

	return 0;
}

```
