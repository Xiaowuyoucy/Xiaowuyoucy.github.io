# 三目运算符的加强

```c++
#include<iostream>

using namespace std;

int main(int argc, char* argv[])
{
	int a, b;
	a = 10;
	b = 20;
	//三目运算符可以用作左值,返回的是a或b的引用
	//c语言则不行,除非这样写(a > b ? &a : &b) = 100;
	(a > b ? a : b) = 100;

	cout << "a = " << a << endl;
	cout << "b = " << b << endl;

	cin.get();

	return 0;
}

```
