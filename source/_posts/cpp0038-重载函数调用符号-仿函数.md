# 重载函数调用符号(仿函数)

![](../../themes/pure/source/images/javawz/.png)

```cpp
#define _CRT_SECURE_NO_WARNINGS
#include<iostream>


using namespace std;

class A
{
public:
	A();
	~A();
	//重载一个伪函数
	int operator()(int value) {
		return value * value;
	}
private:
	int a;
};

A::A()
{
}

A::~A()
{
}

int main(char *argv[], int argc)
{
	A a;

	cout << a(2) << endl;

	return 0;
}

```
